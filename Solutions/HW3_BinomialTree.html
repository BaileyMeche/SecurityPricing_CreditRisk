OptionType = Literal["call", "put"]
def payoff(option_type: OptionType, S, K):
    if option_type == "call":
        return np.maximum(S - K, 0.0)
    else:
        return np.maximum(K - S, 0.0)

def get_option_values(
    S0: float,
    K: float,
    N: int,
    u: float,
    d: float,
    p: float,
    disc: float,
    option_type: OptionType = "call",
    american: bool = False,
) -> float:
    
    # terminal nodes stock values
    j = np.arange(N + 1)
    St = S0 * (u ** j) * (d ** (N - j))
    option_values = payoff(option_type, St, K)

    # backward induction
    for step in range(N - 1, -1, -1):
        option_values = disc * (p * option_values[1:] + (1 - p) * option_values[:-1])
        if american:
            j = np.arange(step + 1)
            St = S0 * (u ** j) * (d ** (step - j))
            exercise = payoff(option_type, St, K)
            option_values = np.maximum(option_values, exercise)

    return float(option_values[0])

def price_cox_ross_rubinstein(
    S0: float,
    K: float,
    r: float,
    q: float,
    sigma: float,
    T: float,
    N: int,
    option_type: OptionType = "call",
    american: bool = False,
) -> float:

    dt = T / N
    u = np.exp(sigma * np.sqrt(dt))
    d = 1.0 / u
    a = np.exp((r - q) * dt)
    p = (a - d) / (u - d)
    p = np.clip(p, 0.0, 1.0)
    disc = np.exp(-r * dt)
    return get_option_values(S0, K, N, u, d, p, disc, option_type, american)

def price_jarrow_rudd(
    S0: float,
    K: float,
    r: float,
    q: float,
    sigma: float,
    T: float,
    N: int,
    option_type: OptionType = "call",
    american: bool = False,
) -> float:

    dt = T / N
    drift = (r - q - 0.5 * sigma**2) * dt
    vol = sigma * np.sqrt(dt)
    u = np.exp(drift + vol)
    d = np.exp(drift - vol)
    p = 0.5
    disc = np.exp(-r * dt)
    return get_option_values(S0, K, N, u, d, p, disc, option_type, american)

# Analysis of European put, near the money

S0,K,r,q,sigma,T,N = 100,95,0.05,0.02,0.25,1,500
bs_price = rg.black_scholes(-1, S0, K, r, T, sigma, divrate=q)['Price']

def make_crr_row(row_n):
    crr_price = price_cox_ross_rubinstein(S0,K,r,q,sigma,T,row_n,"put",False)
    return {"N": row_n, "Tree_N": crr_price, "BS": bs_price, "Error_N": abs(crr_price-bs_price)}

crr_pricing_errors = pd.DataFrame([make_crr_row(n) for n in range(1, N+1)])
crr_pricing_errors["log_N"] = np.log(crr_pricing_errors["N"])
crr_pricing_errors["log_Error_N"] = np.log(crr_pricing_errors["Error_N"])
In[10]:
a, b = np.polyfit(crr_pricing_errors["log_N"], crr_pricing_errors["log_Error_N"], 1)

print("Slope (convergence rate) for Cox-Ross-Rubenstein:", a)

fit_log_error = a*crr_pricing_errors["log_N"] + b
fit_error = np.exp(fit_log_error)

Slope (convergence rate) for Cox-Ross-Rubenstein: -1.004217171618121
